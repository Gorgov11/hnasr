// ------------------------------ UTILITIES ------------------------------ //
function useTheme(): [string, () => void] {
  const [theme, setTheme] = useState<string>(() => {
    const saved = typeof window !== "undefined" ? localStorage.getItem("theme") : null;
    return saved || "dark";
  });
  useEffect(() => {
    const root = window.document.documentElement;
    if (theme === "dark") root.classList.add("dark");
    else root.classList.remove("dark");
    localStorage.setItem("theme", theme);
  }, [theme]);
  const toggle = () => setTheme((t) => (t === "dark" ? "light" : "dark"));
  return [theme, toggle];
}

// ----------------------------- 3D SCENE ------------------------------ //
function TwistyKnot({ color = "#8b5cf6" }) {
  const ref = useRef<any>(null);
  useFrame((_, d) => {
    if (!ref.current) return;
    ref.current.rotation.x += 0.15 * d;
    ref.current.rotation.y += 0.1 * d;
  });
  return (
    <Float floatIntensity={1.5} rotationIntensity={0.6} speed={1}>
      <mesh ref={ref} castShadow receiveShadow>
        <torusKnotGeometry args={[1.2, 0.35, 180, 32]} />
        <meshStandardMaterial color={color} metalness={0.6} roughness={0.25} wireframe />
      </mesh>
    </Float>
  );
}

function Scene() {
  return (
    <>
      <ambientLight intensity={0.5} />
      <directionalLight position={[5, 6, 3]} intensity={1.1} />
      <Stars radius={80} depth={40} count={1200} factor={3} saturation={0} fade speed={0.6} />
      <TwistyKnot />
      <OrbitControls enablePan={false} enableZoom={false} autoRotate autoRotateSpeed={0.6} />
    </>
  );
}
